function reverse(x) {
  let strnum = x.toString().split("").reverse()
  let num = strnum.includes("-")
  if(num == true){
    num2 = strnum.splice(strnum.indexOf("-"))
    strnum.unshift(num2)
    return Number(strnum.join(""))
  } else {
    return Number(strnum.join(""))
  }
}

console.log(reverse(-1234)) // -4321
console.log(reverse(1234)) // 4321
